Main stylesheet oclX-functional.xsl
Documentation

Description

Created on: Jan 1, 2012

Author: Jakub Maly

email: jakub@maly.cz

OclX function library contains definition of functions, which enable the use of OCL iterator expressions in XPath expressions. The library can be also used in Schematron schemas, providing that the Schematron validation process is based on XSLT and uses modified Schematron pipeline XSLTs, which add imports of OclX library.

Look to http://exolutio.com and download eXolutio - a tool which allows translation from OCL scripts to Schematron schemas. You can also find the modified Schematron pipeline XSLTs on this address.

The library uses higher-order functions for evaluation of iterator expressions.

Please, be aware that at this moment, OclX does not support nested collections.

Function oclX:forAllN (collection, body, satisfied)
Documentation

Description

Implements forAll function from OCL. Checks, whether a condition holds for all member of a collection.

The function allows multiple iterators in body.

Results in true if the body expression evaluates to true for each element in the source collection; otherwise, result is false.

Parameters

collection The iterated collection.
body Condition expression evaluated for each member of collection. All parameters and result must be of type boolean.

Return

Returns true, if all members of collection satisfy the condition defined by expression.

<xsl:function name="oclX:forAllN" as="xs:boolean">
  <xsl:param name="collection" as="item()*"/>
  <xsl:param name="body" as="function(*)"/>
  <xsl:variable name="iteratorCount" select="function-arity($body)" as="xs:integer"/>
  <xsl:iterate select="1 to xs:integer(math:pow(count($collection), $iteratorCount))">
    <!-- expect true in the beginning -->
    <xsl:param name="satisfied" as="xs:boolean" select="true()"/>
    <xsl:variable name="indices" select="         oclXin:getIndices(. - 1, count($collection), $iteratorCount, ())"/>
    <xsl:variable name="forThis" select="oclXin:functionItemCall(         $body, $iteratorCount, $indices, $collection)" as="xs:boolean"/>
    <xsl:choose>
      <xsl:when test="$forThis eq false()">
        <xsl:break>
          <xsl:sequence select="false()"/>
        </xsl:break>
      </xsl:when>
      <xsl:otherwise>
        <xsl:next-iteration>
          <xsl:with-param name="satisfied" select="$satisfied and $forThis"/>
        </xsl:next-iteration>
      </xsl:otherwise>
    </xsl:choose>
    <xsl:on-completion>
      <xsl:sequence select="$satisfied"/>
    </xsl:on-completion>
  </xsl:iterate>
</xsl:function>

<xsl:function name="oclX:iterate" as="item()*">
  <xsl:param name="collection" as="item()*"/>
  <xsl:param name="accInit" as="item()*"/>
  <xsl:param name="body" as="function(item(), item()*) as item()*"/>
  <xsl:iterate select="1 to count($collection)">
    <xsl:param name="acc" select="$accInit" as="item()*"/>
    <xsl:if test="$debug">
      <xsl:message>Iteration:
        <xsl:value-of select="."/>/
        <xsl:value-of select="count($collection)"/>
      </xsl:message>
      <xsl:message>Current:
        <xsl:value-of select="$collection[current()]"/>"/></xsl:message>
    </xsl:if>
    <xsl:next-iteration>
      <xsl:with-param name="acc" select="$body($collection[current()], $acc)"/>
    </xsl:next-iteration>
    <xsl:on-completion>
      <xsl:sequence select="$acc"/>
    </xsl:on-completion>
  </xsl:iterate>
</xsl:function>

<xsl:function name="oclX:isUnique" as="item()*">
  <xsl:param name="collection" as="item()*"/>
  <xsl:param name="body" as="function(item()) as item()"/>
  <xsl:variable name="uniqueItems" as="item()*">
    <xsl:sequence select="oclX:iterate(         $collection, (), function($it, $acc) {          $acc, if (oclXin:combined-contains($acc, $body($it))) then () else $body($it)         })"/>

  <xsl:message select="count($collection)"/>
  <xsl:sequence select="if (count($uniqueItems) eq count($collection)) then true() else false()"/>
